* --> todo
? --> no importa con que continua
	del hola????.txt --> eliminar los archivos que empiecen con hola y terminen con .txt
	*.txt --> todos los archivos txt 
	hola.* --> todos los archivos llamados hola, sin importar la extencion
	del *.txt --> eliminar todos los archivos de texto
	del hola.* --> eliminar todos los archivos llamados hola
attrib --> cambiar atributos
+ --> añadir
- --> quitar
R --> lectura
H --> oculto
/S a todos los archivos
/D a todos los directorios

	attrib +R > hola.txt --> añadir lectura unicamente
	attrib +H > hola.txt --> ocultar archivo
	attrib +H > carpeta * /s --> ocultar todos los subarchivos de carpeta
tasklist --> desplegar procesos activos
/IM --> cerrar mediante el nombre
/PID --> cerrar mediante el pid
	taskkill /IM excel.exe --> cerrar excel.exe
	taskkill /PID 9700 --> cerrar el proceso con pid  9700 --> cerrar excel.exe
find --> buscar
	find "Nahuel" Nuevo.txt --> buscar la palabra "Nahuel" en el archivo Nuevo.txt
/I --> ignorar mayusculas y minusculas
| --> combinar comandos 
tasklist | find "chrome" /I --> buscar "Chrome" en el tasklist sin importar las mayusculas o minusculas
clip --> copiar
	ping -4 www.google.com | clip --> opiar el ping de google
	clip < Nuevo.txt --> copiar contenido de Nuevo.txt
